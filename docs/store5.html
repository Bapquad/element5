store5 Object

ExpireData Simulator Methods

store5.open( simulatorType )

Open the cookie database simulator.

How to use:

var cookieDB = store5.open( 'cookie_ds' );
console.log( checkOpera );

CookieDataSim.public( varName[, varValue[, expire]] )

Create the public variable, this variable maybe access everywhere in apps.

  • varName:string - The name of variable.
  • varValue:string - The value of variable.
  • expire:number - In seconds, It is the time which this variable exist. When the time has be over, it has deleted auto.

How to use:

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
cookieDB.public( 'username', 'Cuong H.Vu', 4 * dayUnit );

CookieDataSim.private( varName[, varValue[, expire]] )

Create the private variable, this variable maybe access in current branch only. It meaning user be at root branch will not see this variable, if it has been set up in sub branch.

  • varName:string - The name of variable.
  • varValue:string - The value of variable.
  • expire:number - In seconds, It is the time which this variable exist. When the time has be over, it has deleted auto.

How to use:

In "./sub_branch/index.html"

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
cookieDB.private( 'username', 'Cuong H.Vu', 4 * dayUnit );

In "./index.html"

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
var username = cookieDB.private( 'username' );
console.log( username ); 	// It returns undefined.

CookieDataSim.GetValue()

Get value of cookie variable.

How to use:

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
var cookieItem = cookieDB.public( 'username', 'Cuong H.Vu', 4 * dayUnit );
console.log( cookieItem.GetValue() );

CookieDataSim.SetValue( value )

Set value of cookie variable.

  • varName - The name of variable.

How to use:

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
var cookieItem = cookieDB.public( 'username', 'Cuong H.Vu', 4 * dayUnit );
console.log( cookieItem.SetValue( 'Cuong H.Vu' ) );

CookieDataSim.Remove()

Delete cookie variable.

How to use:

var cookieDB = store5.open( 'cookie_ds' );
var dayUnit = 24 * 60 * 60;
var cookieItem = cookieDB.public( 'username', 'Cuong H.Vu', 4 * dayUnit );
cookieItem.Remove(); 
console.log( cookieItem );

IndexedDB Simulator Methods

IndexedDB.Init( tables )

Initialize a database with tables. this initialize run once when first update or upgrade database;

  • tables - The table structures.

How to use:

var indexDB = store5.open( 'indexed_db' );
indexDB.init(
    { name: 'misson_one', key: 'id', 
        structure: 
        [ 
            { name: 'player_name', key: ['playerName', 'playerBirth'], types: { unique: true } },
        ] 
    }
);

IndexedDB.Close()

Close the database.

How to use:

var indexDB = store5.open( 'indexed_db', 'program' ); // 'program' is an name of database.
indexDB.Init(
    { name: 'misson_one', key: 'id', 
        structure: 
        [ 
            { name: 'player_name', key: ['playerName', 'playerBirth'], types: { unique: true } },
        ] 
    }
); 
indexDB.connect = function() 
{
	indexDB.close( this );
}

IndexedDB.Drop()

Drop the database.

How to use:

var link = store5.open( 'indexed_db', 'program' );
link.Drop();

IndexedDB.Upgrade()

Upgrade the database.

How to use:

var link = store5.open( 'indexed_db', 'program' ); // 'program' is an name of database.
link.Init(
    { name: 'misson_one', key: 'id', 
        structure: 
        [ 
            { name: 'player_name', key: ['playerName', 'playerBirth'], types: { unique: true } },
        ] 
    }
); 
link.connect = function() 
{
    console.log( this );			 // show database object
    console.log( this.From( 'misson_one' ) );    // show table object
    link.Upgrade(
        { name: 'misson_two', key: 'id', 
            structure: 
            [ 
                { name: 'player_name', key: ['playerName', 'playerBirth'], types: { unique: true } },
            ] 
        }
    ); 
}
